SQL CREATE TABLES
------------------

CREATE TABLE Professor (
profID INT NOT NULL PRIMARY KEY
	GENERATED ALWAYS AS IDENTITY
	(START WITH 1, INCREMENT BY 1),
firstName varchar(50),
lastName varchar(50),
email varchar(50),
username varchar(50),
password varchar(50)
);
CREATE TABLE  Program(
progId INT NOT NULL PRIMARY KEY
		GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),
progName varchar(50)
);

CREATE TABLE  Student(
studId INT NOT NULL PRIMARY KEY
		GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),
firstName varchar(50),
lastName varchar(50),
email varchar(50),
username varchar(50),
password varchar(50),
progId INT NOT NULL,
FOREIGN KEY (progId) REFERENCES Program(progId)
);

CREATE TABLE  Course(
courseId INT NOT NULL PRIMARY KEY
		GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),
profId INT NOT NULL,
progId INT NOT NULL,
courseName varchar(50),
courseCode varchar(50),
courseTime varchar(50),
roomNum varchar(50),
FOREIGN KEY (profId) REFERENCES Professor(profId),
FOREIGN KEY (progId) REFERENCES Program(progId)
);

CREATE TABLE Material (
matId INT NOT NULL PRIMARY KEY
		GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),
matType varchar(50),
matName varchar(50),
matWeight varchar(50),
courseId INT NOT NULL,
studId INT NOT NULL,
grade REAL,
FOREIGN KEY (courseId) REFERENCES Course(courseId),
FOREIGN KEY (studId) REFERENCES Student(studId)
);
